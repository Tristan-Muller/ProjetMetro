fopen(fichier, mode d'ouverture) :  r   (lecture seule)
                                    w   (écriture seule)
                                    a   (ajout fin fichier)
                                    r+  (Lecture/ecriture, doit exister)
                                    w+  (lecture/ecriture, supp contenu)
                                    a+  (ajout lecture/ecrituer, fin de fichier)

fclose(fichier) (ferme le fichier ouvert)

feof() : teste la fin d'un fichier  
        EOF : end of file


//Lecture : 
    fgetc() : lire un caractère
    fgets(<chaine>, <taille_chaine>, <fichier>) : lit une ligne
    fscanf(<fichier>,<format>, ...)             : lit du txt formaté


//Ecriture : 
    fputc(<caractère>,<fichier>)        : ecrit un caractère
    fputs(<chaine>, <fichier>)          : ecrit une chaine (une ligne de txt)  (crée le fichier si !existe)
    fprintf(<fichier>, <format>, ...)   : écrit du txt formaté 



//Positionnement : 
    ftell(<fichier>) : retourn la position du curseur dans le fichier
    fseek(<fichier>, <deplacement>, <origine>) : déplace le curseur 
        -> origine :    SEEK_SET (debut de fichier)
                        SEEK_CUR (position courante)
                        SEEK_END (fin du fichier)
    rewind(<fichier>) : réinitialise la position du curseur
